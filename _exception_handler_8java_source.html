57 public void uncaughtException(@NonNull Thread t, Throwable e) {
58 Logger.d(TAG, "Uncaught exception being tracked...");
59
60 // Ensure message is not-null/empty
61 String message = truncateString(e.getMessage(), MAX_MESSAGE_LENGTH);
62 if (message == null || message.isEmpty()) {
63 message = "Android Exception. Null or empty message found";
64 }
65
66 String stack = truncateString(Util.stackTraceToString(e), MAX_STACK_LENGTH);
67 String threadName = truncateString(t.getName(), MAX_THREAD_NAME_LENGTH);
68
69 Integer lineNumber = null;
70 String className = null;
71 if (e.getStackTrace().length > 0) {
72 StackTraceElement stackElement = e.getStackTrace()[0];
73
74 // Ensure lineNumber is greater than or equal to zero
75 lineNumber = stackElement.getLineNumber();
76 if (lineNumber < 0) {
77 lineNumber = null;
78 }
79
80 className = truncateString(stackElement.getClassName(), MAX_CLASS_NAME_LENGTH);
81 }
82
83 String exceptionName = truncateString(e.getClass().getName(), MAX_EXCEPTION_NAME_LENGTH);
84
85 Map<String, Object> data = new HashMap<>();
86 Util.addToMap(Parameters.APP_ERROR_MESSAGE, message, data);
87 Util.addToMap(Parameters.APP_ERROR_STACK, stack, data);
88 Util.addToMap(Parameters.APP_ERROR_THREAD_NAME, threadName, data);
89 Util.addToMap(Parameters.APP_ERROR_THREAD_ID, t.getId(), data);
90 Util.addToMap(Parameters.APP_ERROR_LANG, "JAVA", data);
91 Util.addToMap(Parameters.APP_ERROR_LINE, lineNumber, data);
92 Util.addToMap(Parameters.APP_ERROR_CLASS_NAME, className, data);
93 Util.addToMap(Parameters.APP_ERROR_EXCEPTION_NAME, exceptionName, data);
94 Util.addToMap(Parameters.APP_ERROR_FATAL, true, data);
95
96 SelfDescribing event = new SelfDescribing(new SelfDescribingJson(TrackerConstants.APPLICATION_ERROR_SCHEMA, data));
97 Map<String, Object> notificationData = new HashMap<>();
98 notificationData.put("event", event);
99 NotificationCenter.postNotification("SnowplowCrashReporting", notificationData);
100
101 defaultHandler.uncaughtException(t, e);
102 }
