244 public List<RequestResult> sendRequests(@NonNull List<Request> requests) {
245 List<Future> futures = new ArrayList<>();
246 List<RequestResult> results = new ArrayList<>();
247
248 // Start all requests in the ThreadPool
249 for (Request request : requests) {
250 String userAgent = request.customUserAgent != null ? request.customUserAgent : DEFAULT_USER_AGENT;
251
252 okhttp3.Request okHttpRequest = httpMethod == HttpMethod.GET
253 ? buildGetRequest(request, userAgent)
254 : buildPostRequest(request, userAgent);
255
256 futures.add(Executor.futureCallable(getRequestCallable(okHttpRequest)));
257 }
258
259 Logger.d(TAG, "Request Futures: %s", futures.size());
260
261 // Get results of futures
262 // - Wait up to emitTimeout seconds for the request
263 for (int i = 0; i < futures.size(); i++) {
264 int code = -1;
265
266 try {
267 code = (int) futures.get(i).get(emitTimeout, TimeUnit.SECONDS);
268 } catch (InterruptedException ie) {
269 Logger.e(TAG, "Request Future was interrupted: %s", ie.getMessage());
270 } catch (ExecutionException ee) {
271 Logger.e(TAG, "Request Future failed: %s", ee.getMessage());
272 } catch (TimeoutException te) {
273 Logger.e(TAG, "Request Future had a timeout: %s", te.getMessage());
274 }
275
276 Request request = requests.get(i);
277 List<Long> eventIds = request.emitterEventIds;
278 results.add(new RequestResult(code, request.oversize, eventIds));
279 if (request.oversize) {
280 Logger.track(TAG, "Request is oversized for emitter event IDs: %s", eventIds.toString());
281 }
282 }
283 return results;
284 }
