55 synchronized TrackerStateSnapshot trackerStateForProcessedEvent(@NonNull Event event) {
56 if (event instanceof AbstractSelfDescribing) {
57 AbstractSelfDescribing sdEvent = (AbstractSelfDescribing) event;
58 List<StateMachineInterface> stateMachines = new LinkedList<>();
59 List<StateMachineInterface> stateMachinesForSchema = eventSchemaToStateMachine.get(sdEvent.getSchema());
60 if (stateMachinesForSchema != null) {
61 stateMachines.addAll(stateMachinesForSchema);
62 }
63 List<StateMachineInterface> stateMachinesGeneral = eventSchemaToStateMachine.get("*");
64 if (stateMachinesGeneral != null) {
65 stateMachines.addAll(stateMachinesGeneral);
66 }
67 for (StateMachineInterface stateMachine : stateMachines) {
68 String stateIdentifier = stateMachineToIdentifier.get(stateMachine);
69 StateFuture previousStateFuture = trackerState.getStateFuture(stateIdentifier);
70 StateFuture currentStateFuture = new StateFuture(sdEvent, previousStateFuture, stateMachine);
71 trackerState.put(stateIdentifier, currentStateFuture);
72 // TODO: Remove early state computation.
73 /*
74 The early state-computation causes low performance as it's executed synchronously on
75 the track method thread. Ideally, the state computation should be executed only on
76 entities generation or payload updating (outputs). In that case there are two problems
77 to address:
78 - long chains of StateFuture filling the memory (in case the outputs are not generated)
79 - event object reuse by the user (the event object in the StateFuture could be modified
80 externally)
81 Remove the early state-computation only when these two problems are fixed.

83 currentStateFuture.getState(); // Early state-computation
84 }

126 Map<String, Object> payloadValues = stateMachine.payloadValues(event, state);
127 if (payloadValues != null && !event.addPayloadValues(payloadValues)) {
128 failures++;
129 }
