134 private void trackEvent(@NonNull AbstractEvent event, @Nullable String context, @Nullable String[] trackers) throws JSONException {
135 if (context != null) {
136 List<SelfDescribingJson> contextEntities = parseContext(context);
137 if (!contextEntities.isEmpty()) {
138 event.contexts(contextEntities);
139 }
140 }
141
142 if (trackers == null || trackers.length == 0) {
143 TrackerController tracker = Snowplow.getDefaultTracker();
144 if (tracker != null) {
145 tracker.track(event);
146 } else {
147 Logger.e(TAG,"Tracker not initialized.");
148 }
149 } else {
150 for (String namespace : trackers) {
151 TrackerController tracker = Snowplow.getTracker(namespace);
152 if (tracker != null) {
153 tracker.track(event);
154 } else {
155 Logger.e(TAG, String.format("Tracker with namespace %s not found.", namespace));
156 }
157 }
158 }
159 }

161 private List<SelfDescribingJson> parseContext(@NonNull String context) throws JSONException {
162 List<SelfDescribingJson> entities = new ArrayList<>();
163 JSONArray contextJson = new JSONArray(context);
164
165 for (int i = 0; i < contextJson.length(); i++) {
166 JSONObject itemJson = contextJson.getJSONObject(i);
167 Map<String, Object> item = JsonUtils.jsonToMap(itemJson);
168
169 String schema = (String) item.get("schema");
170 Object data = item.get("data");
171
172 if (schema != null && data != null) {
173 entities.add(new SelfDescribingJson(schema, data));
174 }
175 }
176
177 return entities;
178 }
