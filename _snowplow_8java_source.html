285 @NonNull
286 private synchronized static List<String> createTracker(@NonNull Context context, @NonNull List<ConfigurationBundle> bundles) {
287 List<String> namespaces = new ArrayList<>();
288 for (ConfigurationBundle bundle : bundles) {
289 if (bundle.networkConfiguration == null) {
290 // remove tracker if it exists
291 TrackerController tracker = getTracker(bundle.namespace);
292 if (tracker != null) {
293 removeTracker(tracker);
294 }
295 } else {
296 List<Configuration> list = bundle.getConfigurations();
297 Configuration[] array = list.toArray(new Configuration[list.size()]);
298 createTracker(context, bundle.namespace, bundle.networkConfiguration, array);
299 namespaces.add(bundle.namespace);
300 }
301 }
302 return namespaces;
