298 @NonNull
299 private synchronized static List<String> createTracker(@NonNull Context context, @NonNull List<ConfigurationBundle> bundles) {
300 List<String> namespaces = new ArrayList<>();
301 for (ConfigurationBundle bundle : bundles) {
302 if (bundle.networkConfiguration == null) {
303 // remove tracker if it exists
304 TrackerController tracker = getTracker(bundle.namespace);
305 if (tracker != null) {
306 removeTracker(tracker);
307 }
308 } else {
309 List<Configuration> list = bundle.getConfigurations();
310 Configuration[] array = list.toArray(new Configuration[list.size()]);
311 createTracker(context, bundle.namespace, bundle.networkConfiguration, array);
312 namespaces.add(bundle.namespace);
313 }
314 }
315 return namespaces;
316 }
