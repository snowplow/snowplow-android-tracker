105 public static long getUTF8Length(@NonNull String s) {
106 long len = 0;
107 for (int i = 0; i < s.length(); i++) {
108 char code = s.charAt(i);
109 if (code <= 0x7f) {
110 len += 1;
111 } else if (code <= 0x7ff) {
112 len += 2;
113 } else if (code >= 0xd800 && code <= 0xdfff) {
114 // Surrogate pair: These take 4 bytes in UTF-8 and 2 chars in UCS-2
115 // (Assume next char is the other [valid] half and just skip it)
116 len += 4; i++;
117 } else if (code < 0xffff) {
118 len += 3;
119 } else {
120 len += 4;
121 }
122 }
123 return len;
124 }

158 @NonNull
159 public static String joinLongList(@NonNull List<Long> list) {
160 StringBuilder s = new StringBuilder();
161
162 for (int i = 0; i < list.size(); i++) {
163 Long longVal = list.get(i);
164 if (longVal != null) {
165 s.append(list.get(i));
166 if (i < list.size() - 1) {
167 s.append(",");
168 }
169 }
170 }
171
172 if (s.toString().endsWith(",")) {
173 s = new StringBuilder(s.substring(0, s.length() - 1));
174 }
175
176 return s.toString();
177 }

187 @Nullable
188 public static SelfDescribingJson getGeoLocationContext(@NonNull Context context) {
189 Location location = getLastKnownLocation(context);
190
191 if (location != null) {
192 Map<String, Object> pairs = new HashMap<>();
193 addToMap(Parameters.LATITUDE, location.getLatitude(), pairs);
194 addToMap(Parameters.LONGITUDE, location.getLongitude(), pairs);
195 addToMap(Parameters.ALTITUDE, location.getAltitude(), pairs);
196 addToMap(Parameters.LATLONG_ACCURACY, location.getAccuracy(), pairs);
197 addToMap(Parameters.SPEED, location.getSpeed(), pairs);
198 addToMap(Parameters.BEARING, location.getBearing(), pairs);
199 addToMap(Parameters.GEO_TIMESTAMP, System.currentTimeMillis(), pairs);
200
201 if (mapHasKeys(pairs, Parameters.LATITUDE, Parameters.LONGITUDE)) {
202 return new SelfDescribingJson(
203 TrackerConstants.GEOLOCATION_SCHEMA, pairs
204 );
205 }
206 }
207 return null;
208 }

217 @SuppressLint("MissingPermission") // Suppressed as it's caught by SecurityException catch block.
218 @Nullable
219 public static Location getLastKnownLocation(@NonNull Context context) {
220 LocationManager locationManager = (LocationManager) context.getSystemService(Context.LOCATION_SERVICE);
221 Location location = null;
222
223 try {
224 String locationProvider = null;
225 if (locationManager.isProviderEnabled(LocationManager.GPS_PROVIDER)) {
226 locationProvider = LocationManager.GPS_PROVIDER;
227 } else if (locationManager.isProviderEnabled(LocationManager.NETWORK_PROVIDER)) {
228 locationProvider = LocationManager.NETWORK_PROVIDER;
229 } else {
230 List<String> locationProviders = locationManager.getProviders(true);
231 if (locationProviders.size() > 0) {
232 locationProvider = locationProviders.get(0);
233 }
234 }
235
236 if (locationProvider != null && !locationProvider.equals("")) {
237 location = locationManager.getLastKnownLocation(locationProvider);
238 }
239 } catch (SecurityException ex) {
240 Logger.e(TAG, "Exception occurred when retrieving location: %s", ex.toString());
241 }
242
243 return location;
244 }

251 @Nullable
252 static public SelfDescribingJson getApplicationContext(@NonNull Context context) {
253 try {
254 PackageInfo pInfo = context.getPackageManager().getPackageInfo(context.getPackageName(), 0);
255 String versionName = pInfo.versionName;
256 String versionCode;
257 if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.P) {
258 versionCode = String.valueOf(pInfo.getLongVersionCode());
259 } else {
260 versionCode = String.valueOf(pInfo.versionCode);
261 }
262 if (versionName != null) {
263 Map<String, Object> pairs = new HashMap<>();
264 addToMap(Parameters.APP_VERSION, versionName, pairs);
265 addToMap(Parameters.APP_BUILD, versionCode, pairs);
266 return new SelfDescribingJson(
267 TrackerConstants.SCHEMA_APPLICATION, pairs
268 );
269 }
270 } catch (PackageManager.NameNotFoundException e) {
271 Logger.e(TAG, "Failed to find application context: %s", e.getMessage());
272 }
273 return null;
274 }

348 ByteArrayInputStream mem_in = new ByteArrayInputStream(bytes);

351 in.close();
352 mem_in.close();
353 newMap = map;
354 } catch (NullPointerException | ClassNotFoundException | IOException e) {
355 e.printStackTrace();
