46 private static InitializationState initializationState = InitializationState.NONE;

48 public synchronized static void initialize(@NonNull Context context) {
49 if (initializationState == InitializationState.NONE) {
50 initializationState = InitializationState.IN_PROGRESS;
51 // addObserver must execute on the mainThread
52 Handler mainHandler = new Handler(context.getMainLooper());
53 mainHandler.post(new Runnable() {
54 @Override
55 public void run() {
56 try {
57 ProcessLifecycleOwner.get().getLifecycle().addObserver(new ProcessObserver());
58 initializationState = InitializationState.COMPLETE;
59 } catch (NoClassDefFoundError e) {
60 initializationState = InitializationState.NONE;
61 Logger.e(TAG,"Class 'ProcessLifecycleOwner' not found. The tracker can't track lifecycle events.");
62 }
63 }
64 });
65 }
66 }
