78 public static void track(@NonNull String tag, @NonNull String msg, @Nullable Object... args) {
79 Logger.e(tag, msg, args);
80 try {
81 Throwable throwable = null;
82 for (Object arg : args) {
83 if (Throwable.class.isInstance(arg)) {
84 throwable = (Throwable) arg;
85 break;
86 }
87 }
88 TrackerError event = new TrackerError(tag, getMessage(msg, args), throwable);
89 Map<String, Object> notificationData = new HashMap<String, Object>();
90 notificationData.put("event", event);
91 NotificationCenter.postNotification("SnowplowTrackerDiagnostic", notificationData);
92 } catch (Exception e) {
93 Logger.v(TAG, "Error logger can't report the error: " + e);
94 }
95 }
