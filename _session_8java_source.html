118 @SuppressLint("ApplySharedPref")
119 public Session(long foregroundTimeout, long backgroundTimeout, @NonNull TimeUnit timeUnit, @Nullable String namespace, @NonNull Context context) {
120 this.foregroundTimeout = timeUnit.toMillis(foregroundTimeout);
121 this.backgroundTimeout = timeUnit.toMillis(backgroundTimeout);
122 isSessionCheckerEnabled = true;
123
124 String sessionVarsName = TrackerConstants.SNOWPLOW_SESSION_VARS;
125 if (namespace != null && !namespace.isEmpty()) {
126 String sessionVarsSuffix = namespace.replaceAll("[^a-zA-Z0-9_]+", "-");
127 sessionVarsName = TrackerConstants.SNOWPLOW_SESSION_VARS + "_" + sessionVarsSuffix;
128 }
129
130 StrictMode.ThreadPolicy oldPolicy = StrictMode.allowThreadDiskReads();
131 try {
132 Map<String, Object> sessionInfo = getSessionMapFromLegacyTrackerV3(context, sessionVarsName);
133 if (sessionInfo == null) {
134 sessionInfo = getSessionMapFromLegacyTrackerV2(context, sessionVarsName);
135 if (sessionInfo == null) {
136 try {
137 sessionInfo = getSessionMapFromLegacyTrackerV1(context);
138 } catch (Exception e) {
139 Logger.track(TAG, String.format("Exception occurred retrieving session info from file: %s", e), e);
140 }
141 }
142 }
143 if (sessionInfo == null) {
144 Logger.track(TAG, "No previous session info available");
145 } else {
146 state = SessionState.build(sessionInfo);
147 }
148 userId = retrieveUserId(context, state);
149 sharedPreferences = context.getSharedPreferences(sessionVarsName, Context.MODE_PRIVATE);
150 lastSessionCheck = System.currentTimeMillis();
151 } finally {
152 StrictMode.setThreadPolicy(oldPolicy);
153 }
154 Logger.v(TAG, "Tracker Session Object created.");
155 }

157 private synchronized static String retrieveUserId(Context context, SessionState state) {
158 String userId = state != null ? state.getUserId() : Util.getUUIDString();
159 // Session_UserID is available only if the session context is enabled.
160 // In a future version we would like to make it available even if the session context is disabled.
161 // For this reason, we store the Session_UserID in a separate storage (decoupled by session values)
162 // calling it Installation_UserID in order to remark that it isn't related to the session context.
163 // Although, for legacy, we need to copy its value in the Session_UserID of the session context
164 // as the session context schema (and related data modelling) requires it.
165 // For further details: https://discourse.snowplowanalytics.com/t/rfc-mobile-trackers-v2-0
166 SharedPreferences generalPref = context.getSharedPreferences(TrackerConstants.SNOWPLOW_GENERAL_VARS, Context.MODE_PRIVATE);
167 String storedUserId = generalPref.getString(TrackerConstants.INSTALLATION_USER_ID, null);
168 if (storedUserId != null) {
169 userId = storedUserId;
170 } else {
171 generalPref.edit()
172 .putString(TrackerConstants.INSTALLATION_USER_ID, userId)
173 .commit();
174 }
175 return userId;
176 }

183 @NonNull
184 public synchronized SelfDescribingJson getSessionContext(@NonNull String eventId, long eventTimestamp) {
185 Logger.v(TAG, "Getting session context...");
186 if (isSessionCheckerEnabled) {
187 if (shouldUpdateSession()) {
188 Logger.d(TAG, "Update session information.");
189 updateSession(eventId, eventTimestamp);
190
191 if (isBackground.get()) { // timed out in background
192 this.executeEventCallback(backgroundTimeoutCallback);
193 } else { // timed out in foreground
194 this.executeEventCallback(foregroundTimeoutCallback);
195 }
196 }
197 lastSessionCheck = System.currentTimeMillis();
198 }
199 eventIndex += 1;
200
201 Map<String, Object> sessionValues = state.getSessionValues();
202 Map<String, Object> sessionCopy = new HashMap<>();
203 sessionCopy.putAll(sessionValues);
204 sessionCopy.put(Parameters.SESSION_EVENT_INDEX, eventIndex);
205
206 return new SelfDescribingJson(TrackerConstants.SESSION_SCHEMA, sessionCopy);
207 }

209 private boolean shouldUpdateSession() {
210 if (isNewSession.get()) {
211 return true;
212 }
213 long now = System.currentTimeMillis();
214 long timeout = isBackground.get() ? backgroundTimeout : foregroundTimeout;
215 return now < lastSessionCheck || now - lastSessionCheck > timeout;
216 }

218 private synchronized void updateSession(String eventId, long eventTimestamp) {
219 isNewSession.set(false);
220 String currentSessionId = Util.getUUIDString();
221 String eventTimestampDateTime = Util.getDateTimeFromTimestamp(eventTimestamp);
222
223 int sessionIndex = 1;
224 eventIndex = 0;
225 String previousSessionId = null;
226 String storage = "LOCAL_STORAGE";
227 if (state != null) {
228 sessionIndex = state.getSessionIndex() + 1;
229 previousSessionId = state.getSessionId();
230 storage = state.getStorage();
231 }
232 state = new SessionState(eventId, eventTimestampDateTime, currentSessionId, previousSessionId, sessionIndex, userId, storage);
233 storeSessionState(state);
234 callOnSessionUpdateCallback(state);
235 }

276 public void setBackground(boolean isBackground) {
277 if (!this.isBackground.compareAndSet(!isBackground, isBackground)) {
278 return;
279 }
280
281 if (!isBackground) {
282 Logger.d(TAG, "Application moved to foreground");
283 this.executeEventCallback(this.foregroundTransitionCallback);
284 try {
285 setIsSuspended(false);
286 } catch (Exception e) {
287 Logger.e(TAG, "Could not resume checking as tracker not setup. Exception: %s", e);
288 }
289 foregroundIndex++;
290 } else {
291 Logger.d(TAG, "Application moved to background");
292 this.executeEventCallback(this.backgroundTransitionCallback);
293 backgroundIndex++;
294 }
295 }

337 @Nullable
338 private Map<String, Object> getSessionMapFromLegacyTrackerV2(@NonNull Context context, @NonNull String sessionVarsName) {
339 StrictMode.ThreadPolicy oldPolicy = StrictMode.allowThreadDiskReads();
340 try {
341 SharedPreferences sharedPreferences = context.getSharedPreferences(sessionVarsName, Context.MODE_PRIVATE);
342 if (!sharedPreferences.contains(Parameters.SESSION_ID)) {
343 sharedPreferences = context.getSharedPreferences(TrackerConstants.SNOWPLOW_SESSION_VARS, Context.MODE_PRIVATE);
344 if (!sharedPreferences.contains(Parameters.SESSION_ID)) {
345 return null;
346 }
347 }
348 // Create map used as initial session state
349 Map<String, Object> sessionMap = new HashMap<>();
350 String sessionId = sharedPreferences.getString(Parameters.SESSION_ID, null);
351 if (sessionId == null) return null;
352 sessionMap.put(Parameters.SESSION_ID, sessionId);
353
354 String userId = sharedPreferences.getString(Parameters.SESSION_USER_ID, null);
355 if (userId == null) return null;
356 sessionMap.put(Parameters.SESSION_USER_ID, userId);
357
358 int sessionIndex = sharedPreferences.getInt(Parameters.SESSION_INDEX, 0);
359 sessionMap.put(Parameters.SESSION_INDEX, sessionIndex);
360
361 sessionMap.put(Parameters.SESSION_FIRST_ID, "");
362 sessionMap.put(Parameters.SESSION_PREVIOUS_ID, null);
363 sessionMap.put(Parameters.SESSION_STORAGE, "LOCAL_STORAGE");
364 return sessionMap;
365 } finally {
366 StrictMode.setThreadPolicy(oldPolicy);
367 }
368 }

370 @Nullable
371 private Map<String, Object> getSessionMapFromLegacyTrackerV3(@NonNull Context context, @NonNull String sessionVarsName) {
372 StrictMode.ThreadPolicy oldPolicy = StrictMode.allowThreadDiskReads();
373 try {
374 SharedPreferences sharedPreferences = context.getSharedPreferences(sessionVarsName, Context.MODE_PRIVATE);
375 if (!sharedPreferences.contains(TrackerConstants.SESSION_STATE)) {
376 return null;
377 }
378 Map<String, Object> sessionMap = new HashMap<>();
379 String jsonString = sharedPreferences.getString(TrackerConstants.SESSION_STATE, null);
380 JSONObject jsonObject = new JSONObject(jsonString);
381 Iterator<String> iterator = jsonObject.keys();
382 while (iterator.hasNext()) {
383 String key = iterator.next();
384 Object value = jsonObject.get(key);
385 sessionMap.put(key, value);
386 }
387 return sessionMap;
388 } catch (JSONException e) {
389 e.printStackTrace();
390 } finally {
391 StrictMode.setThreadPolicy(oldPolicy);
392 }
393 return null;
394 }
