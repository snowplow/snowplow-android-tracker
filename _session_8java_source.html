117 @SuppressLint("ApplySharedPref")
118 public Session(long foregroundTimeout, long backgroundTimeout, @NonNull TimeUnit timeUnit, @Nullable String namespace, @NonNull Context context) {
119 this.foregroundTimeout = timeUnit.toMillis(foregroundTimeout);
120 this.backgroundTimeout = timeUnit.toMillis(backgroundTimeout);
121 isSessionCheckerEnabled = true;
122
123 String sessionVarsName = TrackerConstants.SNOWPLOW_SESSION_VARS;
124 if (namespace != null && !namespace.isEmpty()) {
125 String sessionVarsSuffix = namespace.replaceAll("[^a-zA-Z0-9_]+", "-");
126 sessionVarsName = TrackerConstants.SNOWPLOW_SESSION_VARS + "_" + sessionVarsSuffix;
127 }
128
129 StrictMode.ThreadPolicy oldPolicy = StrictMode.allowThreadDiskReads();
130 try {
131 Map<String, Object> sessionInfo = getSessionMapFromLegacyTrackerV3(context, sessionVarsName);
132 if (sessionInfo == null) {
133 sessionInfo = getSessionMapFromLegacyTrackerV2(context, sessionVarsName);
134 if (sessionInfo == null) {
135 try {
136 sessionInfo = getSessionMapFromLegacyTrackerV1(context);
137 } catch (Exception e) {
138 Logger.track(TAG, String.format("Exception occurred retrieving session info from file: %s", e), e);
139 }
140 }
141 }
142 if (sessionInfo == null) {
143 Logger.track(TAG, "No previous session info available");
144 } else {
145 state = SessionState.build(sessionInfo);
146 }
147 userId = retrieveUserId(context, state);
148 sharedPreferences = context.getSharedPreferences(sessionVarsName, Context.MODE_PRIVATE);
149 lastSessionCheck = System.currentTimeMillis();
150 } finally {
151 StrictMode.setThreadPolicy(oldPolicy);
152 }
153 Logger.v(TAG, "Tracker Session Object created.");
154 }

156 private synchronized static String retrieveUserId(Context context, SessionState state) {
157 String userId = state != null ? state.getUserId() : Util.getUUIDString();
158 // Session_UserID is available only if the session context is enabled.
159 // In a future version we would like to make it available even if the session context is disabled.
160 // For this reason, we store the Session_UserID in a separate storage (decoupled by session values)
161 // calling it Installation_UserID in order to remark that it isn't related to the session context.
162 // Although, for legacy, we need to copy its value in the Session_UserID of the session context
163 // as the session context schema (and related data modelling) requires it.
164 // For further details: https://discourse.snowplowanalytics.com/t/rfc-mobile-trackers-v2-0
165 SharedPreferences generalPref = context.getSharedPreferences(TrackerConstants.SNOWPLOW_GENERAL_VARS, Context.MODE_PRIVATE);
166 String storedUserId = generalPref.getString(TrackerConstants.INSTALLATION_USER_ID, null);
167 if (storedUserId != null) {
168 userId = storedUserId;
169 } else {
170 generalPref.edit()
171 .putString(TrackerConstants.INSTALLATION_USER_ID, userId)
172 .commit();
173 }
174 return userId;
175 }

182 @NonNull
183 public synchronized SelfDescribingJson getSessionContext(@NonNull String eventId) {
184 Logger.v(TAG, "Getting session context...");
185 if (isSessionCheckerEnabled) {
186 if (shouldUpdateSession()) {
187 Logger.d(TAG, "Update session information.");
188 updateSession(eventId);
189
190 if (isBackground.get()) { // timed out in background
191 this.executeEventCallback(backgroundTimeoutCallback);
192 } else { // timed out in foreground
193 this.executeEventCallback(foregroundTimeoutCallback);
194 }
195 }
196 lastSessionCheck = System.currentTimeMillis();
197 }
198 return new SelfDescribingJson(TrackerConstants.SESSION_SCHEMA, state.getSessionValues());
199 }

201 private boolean shouldUpdateSession() {
202 if (isNewSession.get()) {
203 return true;
204 }
205 long now = System.currentTimeMillis();
206 long timeout = isBackground.get() ? backgroundTimeout : foregroundTimeout;
207 return now < lastSessionCheck || now - lastSessionCheck > timeout;
208 }

210 private synchronized void updateSession(String eventId) {
211 isNewSession.set(false);
212 String currentSessionId = Util.getUUIDString();
213 int sessionIndex = 1;
214 String previousSessionId = null;
215 String storage = "LOCAL_STORAGE";
216 if (state != null) {
217 sessionIndex = state.getSessionIndex() + 1;
218 previousSessionId = state.getSessionId();
219 storage = state.getStorage();
220 }
221 state = new SessionState(eventId, currentSessionId, previousSessionId, sessionIndex, userId, storage);
222 storeSessionState(state);
223 callOnSessionUpdateCallback(state);
224 }

265 public void setBackground(boolean isBackground) {
266 if (!this.isBackground.compareAndSet(!isBackground, isBackground)) {
267 return;
268 }
269
270 if (!isBackground) {
271 Logger.d(TAG, "Application moved to foreground");
272 this.executeEventCallback(this.foregroundTransitionCallback);
273 try {
274 setIsSuspended(false);
275 } catch (Exception e) {
276 Logger.e(TAG, "Could not resume checking as tracker not setup. Exception: %s", e);
277 }
278 foregroundIndex++;
279 } else {
280 Logger.d(TAG, "Application moved to background");
281 this.executeEventCallback(this.backgroundTransitionCallback);
282 backgroundIndex++;
283 }
284 }

326 @Nullable
327 private Map<String, Object> getSessionMapFromLegacyTrackerV2(@NonNull Context context, @NonNull String sessionVarsName) {
328 StrictMode.ThreadPolicy oldPolicy = StrictMode.allowThreadDiskReads();
329 try {
330 SharedPreferences sharedPreferences = context.getSharedPreferences(sessionVarsName, Context.MODE_PRIVATE);
331 if (!sharedPreferences.contains(Parameters.SESSION_ID)) {
332 sharedPreferences = context.getSharedPreferences(TrackerConstants.SNOWPLOW_SESSION_VARS, Context.MODE_PRIVATE);
333 if (!sharedPreferences.contains(Parameters.SESSION_ID)) {
334 return null;
335 }
336 }
337 // Create map used as initial session state
338 Map<String, Object> sessionMap = new HashMap<>();
339 String sessionId = sharedPreferences.getString(Parameters.SESSION_ID, null);
340 if (sessionId == null) return null;
341 sessionMap.put(Parameters.SESSION_ID, sessionId);
342
343 String userId = sharedPreferences.getString(Parameters.SESSION_USER_ID, null);
344 if (userId == null) return null;
345 sessionMap.put(Parameters.SESSION_USER_ID, userId);
346
347 int sessionIndex = sharedPreferences.getInt(Parameters.SESSION_INDEX, 0);
348 sessionMap.put(Parameters.SESSION_INDEX, sessionIndex);
349
350 sessionMap.put(Parameters.SESSION_FIRST_ID, "");
351 sessionMap.put(Parameters.SESSION_PREVIOUS_ID, null);
352 sessionMap.put(Parameters.SESSION_STORAGE, "LOCAL_STORAGE");
353 return sessionMap;
354 } finally {
355 StrictMode.setThreadPolicy(oldPolicy);
356 }
357 }

359 @Nullable
360 private Map<String, Object> getSessionMapFromLegacyTrackerV3(@NonNull Context context, @NonNull String sessionVarsName) {
361 StrictMode.ThreadPolicy oldPolicy = StrictMode.allowThreadDiskReads();
362 try {
363 SharedPreferences sharedPreferences = context.getSharedPreferences(sessionVarsName, Context.MODE_PRIVATE);
364 if (!sharedPreferences.contains(TrackerConstants.SESSION_STATE)) {
365 return null;
366 }
367 Map<String, Object> sessionMap = new HashMap<>();
368 String jsonString = sharedPreferences.getString(TrackerConstants.SESSION_STATE, null);
369 JSONObject jsonObject = new JSONObject(jsonString);
370 Iterator<String> iterator = jsonObject.keys();
371 while (iterator.hasNext()) {
372 String key = iterator.next();
373 Object value = jsonObject.get(key);
374 sessionMap.put(key, value);
375 }
376 return sessionMap;
377 } catch (JSONException e) {
378 e.printStackTrace();
379 } finally {
380 StrictMode.setThreadPolicy(oldPolicy);
381 }
382 return null;
383 }
