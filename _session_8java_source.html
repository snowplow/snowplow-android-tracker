119 @SuppressLint("ApplySharedPref")
120 public Session(long foregroundTimeout, long backgroundTimeout, @NonNull TimeUnit timeUnit, @Nullable String namespace, @NonNull Context context) {
121 this.foregroundTimeout = timeUnit.toMillis(foregroundTimeout);
122 this.backgroundTimeout = timeUnit.toMillis(backgroundTimeout);
123 isSessionCheckerEnabled = true;
124
125 String sessionVarsName = TrackerConstants.SNOWPLOW_SESSION_VARS;
126 if (namespace != null && !namespace.isEmpty()) {
127 String sessionVarsSuffix = namespace.replaceAll("[^a-zA-Z0-9_]+", "-");
128 sessionVarsName = TrackerConstants.SNOWPLOW_SESSION_VARS + "_" + sessionVarsSuffix;
129 }
130
131 StrictMode.ThreadPolicy oldPolicy = StrictMode.allowThreadDiskReads();
132 try {
133 Map<String, Object> sessionInfo = getSessionMapFromLegacyTrackerV3(context, sessionVarsName);
134 if (sessionInfo == null) {
135 sessionInfo = getSessionMapFromLegacyTrackerV2(context, sessionVarsName);
136 if (sessionInfo == null) {
137 try {
138 sessionInfo = getSessionMapFromLegacyTrackerV1(context);
139 } catch (Exception e) {
140 Logger.track(TAG, String.format("Exception occurred retrieving session info from file: %s", e), e);
141 }
142 }
143 }
144 if (sessionInfo == null) {
145 Logger.track(TAG, "No previous session info available");
146 } else {
147 state = SessionState.build(sessionInfo);
148 }
149 userId = retrieveUserId(context, state);
150 sharedPreferences = context.getSharedPreferences(sessionVarsName, Context.MODE_PRIVATE);
151 lastSessionCheck = System.currentTimeMillis();
152 } finally {
153 StrictMode.setThreadPolicy(oldPolicy);
154 }
155 Logger.v(TAG, "Tracker Session Object created.");
156 }

158 private synchronized static String retrieveUserId(Context context, SessionState state) {
159 String userId = state != null ? state.getUserId() : Util.getUUIDString();
160 // Session_UserID is available only if the session context is enabled.
161 // In a future version we would like to make it available even if the session context is disabled.
162 // For this reason, we store the Session_UserID in a separate storage (decoupled by session values)
163 // calling it Installation_UserID in order to remark that it isn't related to the session context.
164 // Although, for legacy, we need to copy its value in the Session_UserID of the session context
165 // as the session context schema (and related data modelling) requires it.
166 // For further details: https://discourse.snowplow.io/t/rfc-mobile-trackers-v2-0
167 SharedPreferences generalPref = context.getSharedPreferences(TrackerConstants.SNOWPLOW_GENERAL_VARS, Context.MODE_PRIVATE);
168 String storedUserId = generalPref.getString(TrackerConstants.INSTALLATION_USER_ID, null);
169 if (storedUserId != null) {
170 userId = storedUserId;
171 } else {
172 generalPref.edit()
173 .putString(TrackerConstants.INSTALLATION_USER_ID, userId)
174 .commit();
175 }
176 return userId;
177 }

184 @NonNull
185 public synchronized SelfDescribingJson getSessionContext(@NonNull String eventId, long eventTimestamp, boolean userAnonymisation) {
186 Logger.v(TAG, "Getting session context...");
187 if (isSessionCheckerEnabled) {
188 if (shouldUpdateSession()) {
189 Logger.d(TAG, "Update session information.");
190 updateSession(eventId, eventTimestamp);
191
192 if (isBackground.get()) { // timed out in background
193 this.executeEventCallback(backgroundTimeoutCallback);
194 } else { // timed out in foreground
195 this.executeEventCallback(foregroundTimeoutCallback);
196 }
197 }
198 lastSessionCheck = System.currentTimeMillis();
199 }
200 eventIndex += 1;
201
202 Map<String, Object> sessionValues = state.getSessionValues();
203 Map<String, Object> sessionCopy = new HashMap<>(sessionValues);
204 sessionCopy.put(Parameters.SESSION_EVENT_INDEX, eventIndex);
205 if (userAnonymisation) {
206 sessionCopy.put(Parameters.SESSION_USER_ID, new UUID(0, 0).toString());
207 sessionCopy.put(Parameters.SESSION_PREVIOUS_ID, null);
208 }
209
210 return new SelfDescribingJson(TrackerConstants.SESSION_SCHEMA, sessionCopy);
211 }

213 private boolean shouldUpdateSession() {
214 if (isNewSession.get()) {
215 return true;
216 }
217 long now = System.currentTimeMillis();
218 long timeout = isBackground.get() ? backgroundTimeout : foregroundTimeout;
219 return now < lastSessionCheck || now - lastSessionCheck > timeout;
220 }

222 private synchronized void updateSession(String eventId, long eventTimestamp) {
223 isNewSession.set(false);
224 String currentSessionId = Util.getUUIDString();
225 String eventTimestampDateTime = Util.getDateTimeFromTimestamp(eventTimestamp);
226
227 int sessionIndex = 1;
228 eventIndex = 0;
229 String previousSessionId = null;
230 String storage = "LOCAL_STORAGE";
231 if (state != null) {
232 sessionIndex = state.getSessionIndex() + 1;
233 previousSessionId = state.getSessionId();
234 storage = state.getStorage();
235 }
236 state = new SessionState(eventId, eventTimestampDateTime, currentSessionId, previousSessionId, sessionIndex, userId, storage);
237 storeSessionState(state);
238 callOnSessionUpdateCallback(state);
239 }

280 public void setBackground(boolean isBackground) {
281 if (!this.isBackground.compareAndSet(!isBackground, isBackground)) {
282 return;
283 }
284
285 if (!isBackground) {
286 Logger.d(TAG, "Application moved to foreground");
287 this.executeEventCallback(this.foregroundTransitionCallback);
288 try {
289 setIsSuspended(false);
290 } catch (Exception e) {
291 Logger.e(TAG, "Could not resume checking as tracker not setup. Exception: %s", e);
292 }
293 foregroundIndex++;
294 } else {
295 Logger.d(TAG, "Application moved to background");
296 this.executeEventCallback(this.backgroundTransitionCallback);
297 backgroundIndex++;
298 }
299 }

341 @Nullable
342 private Map<String, Object> getSessionMapFromLegacyTrackerV2(@NonNull Context context, @NonNull String sessionVarsName) {
343 StrictMode.ThreadPolicy oldPolicy = StrictMode.allowThreadDiskReads();
344 try {
345 SharedPreferences sharedPreferences = context.getSharedPreferences(sessionVarsName, Context.MODE_PRIVATE);
346 if (!sharedPreferences.contains(Parameters.SESSION_ID)) {
347 sharedPreferences = context.getSharedPreferences(TrackerConstants.SNOWPLOW_SESSION_VARS, Context.MODE_PRIVATE);
348 if (!sharedPreferences.contains(Parameters.SESSION_ID)) {
349 return null;
350 }
351 }
352 // Create map used as initial session state
353 Map<String, Object> sessionMap = new HashMap<>();
354 String sessionId = sharedPreferences.getString(Parameters.SESSION_ID, null);
355 if (sessionId == null) return null;
356 sessionMap.put(Parameters.SESSION_ID, sessionId);
357
358 String userId = sharedPreferences.getString(Parameters.SESSION_USER_ID, null);
359 if (userId == null) return null;
360 sessionMap.put(Parameters.SESSION_USER_ID, userId);
361
362 int sessionIndex = sharedPreferences.getInt(Parameters.SESSION_INDEX, 0);
363 sessionMap.put(Parameters.SESSION_INDEX, sessionIndex);
364
365 sessionMap.put(Parameters.SESSION_FIRST_ID, "");
366 sessionMap.put(Parameters.SESSION_PREVIOUS_ID, null);
367 sessionMap.put(Parameters.SESSION_STORAGE, "LOCAL_STORAGE");
368 return sessionMap;
369 } finally {
370 StrictMode.setThreadPolicy(oldPolicy);
371 }
372 }

374 @Nullable
375 private Map<String, Object> getSessionMapFromLegacyTrackerV3(@NonNull Context context, @NonNull String sessionVarsName) {
376 StrictMode.ThreadPolicy oldPolicy = StrictMode.allowThreadDiskReads();
377 try {
378 SharedPreferences sharedPreferences = context.getSharedPreferences(sessionVarsName, Context.MODE_PRIVATE);
379 if (!sharedPreferences.contains(TrackerConstants.SESSION_STATE)) {
380 return null;
381 }
382 Map<String, Object> sessionMap = new HashMap<>();
383 String jsonString = sharedPreferences.getString(TrackerConstants.SESSION_STATE, null);
384 JSONObject jsonObject = new JSONObject(jsonString);
385 Iterator<String> iterator = jsonObject.keys();
386 while (iterator.hasNext()) {
387 String key = iterator.next();
388 Object value = jsonObject.get(key);
389 sessionMap.put(key, value);
390 }
391 return sessionMap;
392 } catch (JSONException e) {
393 e.printStackTrace();
394 } finally {
395 StrictMode.setThreadPolicy(oldPolicy);
396 }
397 return null;
398 }
