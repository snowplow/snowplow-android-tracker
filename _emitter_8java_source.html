327 public Emitter(@NonNull Context context, @NonNull String collectorUri, @Nullable EmitterBuilder builder) {
328 this.context = context;
329 if (builder == null) {
330 builder = new EmitterBuilder();
331 }
332 this.requestCallback = builder.requestCallback;
333 this.bufferOption = builder.bufferOption;
334 this.requestSecurity = builder.requestSecurity;
335 this.tlsVersions = builder.tlsVersions;
336 this.emitterTick = builder.emitterTick;
337 this.emptyLimit = builder.emptyLimit;
338 this.sendLimit = builder.sendLimit;
339 this.byteLimitGet = builder.byteLimitGet;
340 this.byteLimitPost = builder.byteLimitPost;
341 this.emitTimeout = builder.emitTimeout;
342 this.timeUnit = builder.timeUnit;
343 this.client = builder.client;
344 this.eventStore = builder.eventStore;
345
346 this.uri = collectorUri;
347 this.httpMethod = builder.httpMethod;
348 this.customPostPath = builder.customPostPath;
349 if (builder.networkConnection == null) {
350 isCustomNetworkConnection = false;
351 String endpoint = collectorUri;
352 if (!endpoint.startsWith("http")) {
353 String protocol = builder.requestSecurity == Protocol.HTTPS ? "https://" : "http://";
354 endpoint = protocol + endpoint;
355 }
356 this.uri = endpoint;
357 setNetworkConnection(new OkHttpNetworkConnection.OkHttpNetworkConnectionBuilder(endpoint, context)
358 .method(builder.httpMethod)
359 .tls(builder.tlsVersions)
360 .emitTimeout(builder.emitTimeout)
361 .customPostPath(builder.customPostPath)
362 .client(builder.client)
363 .cookieJar(builder.cookieJar)
364 .build());
365 } else {
366 isCustomNetworkConnection = true;
367 setNetworkConnection(builder.networkConnection);
368 }
369
370 if (builder.threadPoolSize > 2) {
371 Executor.setThreadCount(builder.threadPoolSize);
372 }
373
374 setCustomRetryForStatusCodes(builder.customRetryForStatusCodes);
375
376 Logger.v(TAG, "Emitter created successfully!");
377 }

448 public boolean shutdown(long timeout) {
449 Logger.d(TAG, "Shutting down emitter.");
450 isRunning.compareAndSet(true, false);
451 ExecutorService es = Executor.shutdown();
452 if (es == null || timeout <= 0) {
453 return true;
454 }
455 try {
456 boolean isTerminated = es.awaitTermination(timeout, TimeUnit.SECONDS);
457 Logger.d(TAG, "Executor is terminated: " + isTerminated);
458 return isTerminated;
459 } catch (InterruptedException e) {
460 Logger.e(TAG, "Executor termination is interrupted: " + e.getMessage());
461 return false;
462 }

480 @SuppressWarnings("all")
481 private void attemptEmit(NetworkConnection networkConnection) {
482 if (isEmittingPaused.get()) {
483 Logger.d(TAG, "Emitter paused.");
484 isRunning.compareAndSet(true, false);
485 return;
486 }
487 if (!Util.isOnline(this.context)) {
488 Logger.d(TAG, "Emitter loop stopping: emitter offline.");
489 isRunning.compareAndSet(true, false);
490 return;
491 }
492 if (eventStore.getSize() <= 0) {
493 if (emptyCount >= this.emptyLimit) {
494 Logger.d(TAG, "Emitter loop stopping: empty limit reached.");
495 isRunning.compareAndSet(true, false);
496 return;
497 }
498 emptyCount++;
499 Logger.e(TAG, "Emitter database empty: " + emptyCount);
500 try {
501 this.timeUnit.sleep(this.emitterTick);
502 } catch (InterruptedException e) {
503 Logger.e(TAG, "Emitter thread sleep interrupted: " + e.toString());
504 }
505 attemptEmit(getNetworkConnection()); // at this point we update network connection since it might be outdated after sleep
506 return;
507 }
508 emptyCount = 0;
509
510 List<EmitterEvent> events = eventStore.getEmittableEvents(sendLimit);
511 List<Request> requests = buildRequests(events, networkConnection.getHttpMethod());
512 List<RequestResult> results = networkConnection.sendRequests(requests);
513
514 Logger.v(TAG, "Processing emitter results.");
515
516 int successCount = 0;
517 int failedWillRetryCount = 0;
518 int failedWontRetryCount = 0;
519 List<Long> removableEvents = new ArrayList<>();
520
521 for (RequestResult res : results) {
522 if (res.isSuccessful()) {
523 removableEvents.addAll(res.getEventIds());
524 successCount += res.getEventIds().size();
525 } else if (res.shouldRetry(customRetryForStatusCodes.get())) {
526 failedWillRetryCount += res.getEventIds().size();
527 Logger.e(TAG, "Request sending failed but we will retry later.");
528 } else {
529 failedWontRetryCount += res.getEventIds().size();
530 removableEvents.addAll(res.getEventIds());
531 Logger.e(TAG, String.format("Sending events to Collector failed with status %d. Events will be dropped.", res.getStatusCode()));
532 }
533 }
534 eventStore.removeEvents(removableEvents);
535
536 int allFailureCount = failedWillRetryCount + failedWontRetryCount;
537
538 Logger.d(TAG, "Success Count: %s", successCount);
539 Logger.d(TAG, "Failure Count: %s", allFailureCount);
540
541 if (requestCallback != null) {
542 if (allFailureCount != 0) {
543 requestCallback.onFailure(successCount, allFailureCount);
544 } else {
545 requestCallback.onSuccess(successCount);
546 }
547 }
548
549 if (failedWillRetryCount > 0 && successCount == 0) {
550 if (Util.isOnline(this.context)) {
551 Logger.e(TAG, "Ensure collector path is valid: %s", networkConnection.getUri());
552 }
553 Logger.e(TAG, "Emitter loop stopping: failures.");
554 isRunning.compareAndSet(true, false);
555 } else {
556 attemptEmit(getNetworkConnection()); // refresh network connection for next emit
557 }
558 }

571 @NonNull
572 protected List<Request> buildRequests(@NonNull List<EmitterEvent> events, HttpMethod httpMethod) {
573 List<Request> requests = new ArrayList<>();
574 String sendingTime = Util.getTimestamp();
575
576 if (httpMethod == GET) {
577 for (EmitterEvent event : events) {
578 Payload payload = event.payload;
579 addSendingTimeToPayload(payload, sendingTime);
580 boolean isOversize = isOversize(payload, httpMethod);
581 Request request = new Request(payload, event.eventId, isOversize);
582 requests.add(request);
583 }
584 } else {
585 for (int i = 0; i < events.size(); i += bufferOption.getCode()) {
586 List<Long> reqEventIds = new ArrayList<>();
587 List<Payload> postPayloadMaps = new ArrayList<>();
588
589 for (int j = i; j < (i + bufferOption.getCode()) && j < events.size(); j++) {
590 EmitterEvent event = events.get(j);
591 Payload payload = event.payload;
592 Long eventId = event.eventId;
593 addSendingTimeToPayload(payload, sendingTime);
594
595 if (isOversize(payload, httpMethod)) {
596 Request request = new Request(payload, eventId, true);
597 requests.add(request);
598
599 } else if (isOversize(payload, postPayloadMaps, httpMethod)) {
600 Request request = new Request(postPayloadMaps, reqEventIds);
601 requests.add(request);
602
603 // Clear collections and build a new POST
604 postPayloadMaps = new ArrayList<>();
605 reqEventIds = new ArrayList<>();
606
607 // Build and store the request
608 postPayloadMaps.add(payload);
609 reqEventIds.add(eventId);
610
611 } else {
612 postPayloadMaps.add(payload);
613 reqEventIds.add(eventId);
614 }
615 }
616
617 // Check if all payloads have been processed
618 if (!postPayloadMaps.isEmpty()) {
619 Request request = new Request(postPayloadMaps, reqEventIds);
620 requests.add(request);
621 }
622 }
623 }
624 return requests;
625 }

633 private boolean isOversize(@NonNull Payload payload, HttpMethod httpMethod) {

644 private boolean isOversize(@NonNull Payload payload, @NonNull List<Payload> previousPaylods, HttpMethod httpMethod) {
645 long byteLimit = httpMethod == GET ? byteLimitGet : byteLimitPost;
646 return isOversize(payload, byteLimit, previousPaylods);
647 }
648
656 private boolean isOversize(@NonNull Payload payload, long byteLimit, @NonNull List<Payload> previousPaylods) {
657 long totalByteSize = payload.getByteSize();
658 for (Payload previousPayload : previousPaylods) {
659 totalByteSize += previousPayload.getByteSize();
660 }
661 int wrapperBytes = previousPaylods.size() > 0 ? (previousPaylods.size() + POST_WRAPPER_BYTES) : 0;
662 return totalByteSize + wrapperBytes > byteLimit;
663 }
