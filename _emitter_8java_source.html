340 public Emitter(@NonNull Context context, @NonNull String collectorUri, @Nullable EmitterBuilder builder) {
341 this.context = context;
342 if (builder == null) {
343 builder = new EmitterBuilder();
344 }
345 this.requestCallback = builder.requestCallback;
346 this.bufferOption = builder.bufferOption;
347 this.requestSecurity = builder.requestSecurity;
348 this.tlsVersions = builder.tlsVersions;
349 this.emitterTick = builder.emitterTick;
350 this.emptyLimit = builder.emptyLimit;
351 this.sendLimit = builder.sendLimit;
352 this.byteLimitGet = builder.byteLimitGet;
353 this.byteLimitPost = builder.byteLimitPost;
354 this.emitTimeout = builder.emitTimeout;
355 this.timeUnit = builder.timeUnit;
356 this.client = builder.client;
357 this.eventStore = builder.eventStore;
358 this.serverAnonymisation = builder.serverAnonymisation;
359
360 this.uri = collectorUri;
361 this.httpMethod = builder.httpMethod;
362 this.customPostPath = builder.customPostPath;
363 if (builder.networkConnection == null) {
364 isCustomNetworkConnection = false;
365 String endpoint = collectorUri;
366 if (!endpoint.startsWith("http")) {
367 String protocol = builder.requestSecurity == Protocol.HTTPS ? "https://" : "http://";
368 endpoint = protocol + endpoint;
369 }
370 this.uri = endpoint;
371 setNetworkConnection(new OkHttpNetworkConnection.OkHttpNetworkConnectionBuilder(endpoint, context)
372 .method(builder.httpMethod)
373 .tls(builder.tlsVersions)
374 .emitTimeout(builder.emitTimeout)
375 .customPostPath(builder.customPostPath)
376 .client(builder.client)
377 .cookieJar(builder.cookieJar)
378 .serverAnonymisation(builder.serverAnonymisation)
379 .build());
380 } else {
381 isCustomNetworkConnection = true;
382 setNetworkConnection(builder.networkConnection);
383 }
384
385 if (builder.threadPoolSize > 2) {
386 Executor.setThreadCount(builder.threadPoolSize);
387 }
388
389 setCustomRetryForStatusCodes(builder.customRetryForStatusCodes);
390
391 Logger.v(TAG, "Emitter created successfully!");
392 }

463 public boolean shutdown(long timeout) {
464 Logger.d(TAG, "Shutting down emitter.");
465 isRunning.compareAndSet(true, false);
466 ExecutorService es = Executor.shutdown();
467 if (es == null || timeout <= 0) {
468 return true;
469 }
470 try {
471 boolean isTerminated = es.awaitTermination(timeout, TimeUnit.SECONDS);
472 Logger.d(TAG, "Executor is terminated: " + isTerminated);
473 return isTerminated;
474 } catch (InterruptedException e) {
475 Logger.e(TAG, "Executor termination is interrupted: " + e.getMessage());
476 return false;
477 }

495 @SuppressWarnings("all")
496 private void attemptEmit(NetworkConnection networkConnection) {
497 if (isEmittingPaused.get()) {
498 Logger.d(TAG, "Emitter paused.");
499 isRunning.compareAndSet(true, false);
500 return;
501 }
502 if (!Util.isOnline(this.context)) {
503 Logger.d(TAG, "Emitter loop stopping: emitter offline.");
504 isRunning.compareAndSet(true, false);
505 return;
506 }
507 if (eventStore.getSize() <= 0) {
508 if (emptyCount >= this.emptyLimit) {
509 Logger.d(TAG, "Emitter loop stopping: empty limit reached.");
510 isRunning.compareAndSet(true, false);
511 return;
512 }
513 emptyCount++;
514 Logger.e(TAG, "Emitter database empty: " + emptyCount);
515 try {
516 this.timeUnit.sleep(this.emitterTick);
517 } catch (InterruptedException e) {
518 Logger.e(TAG, "Emitter thread sleep interrupted: " + e.toString());
519 }
520 attemptEmit(getNetworkConnection()); // at this point we update network connection since it might be outdated after sleep
521 return;
522 }
523 emptyCount = 0;
524
525 List<EmitterEvent> events = eventStore.getEmittableEvents(sendLimit);
526 List<Request> requests = buildRequests(events, networkConnection.getHttpMethod());
527 List<RequestResult> results = networkConnection.sendRequests(requests);
528
529 Logger.v(TAG, "Processing emitter results.");
530
531 int successCount = 0;
532 int failedWillRetryCount = 0;
533 int failedWontRetryCount = 0;
534 List<Long> removableEvents = new ArrayList<>();
535
536 for (RequestResult res : results) {
537 if (res.isSuccessful()) {
538 removableEvents.addAll(res.getEventIds());
539 successCount += res.getEventIds().size();
540 } else if (res.shouldRetry(customRetryForStatusCodes.get())) {
541 failedWillRetryCount += res.getEventIds().size();
542 Logger.e(TAG, "Request sending failed but we will retry later.");
543 } else {
544 failedWontRetryCount += res.getEventIds().size();
545 removableEvents.addAll(res.getEventIds());
546 Logger.e(TAG, String.format("Sending events to Collector failed with status %d. Events will be dropped.", res.getStatusCode()));
547 }
548 }
549 eventStore.removeEvents(removableEvents);
550
551 int allFailureCount = failedWillRetryCount + failedWontRetryCount;
552
553 Logger.d(TAG, "Success Count: %s", successCount);
554 Logger.d(TAG, "Failure Count: %s", allFailureCount);
555
556 if (requestCallback != null) {
557 if (allFailureCount != 0) {
558 requestCallback.onFailure(successCount, allFailureCount);
559 } else {
560 requestCallback.onSuccess(successCount);
561 }
562 }
563
564 if (failedWillRetryCount > 0 && successCount == 0) {
565 if (Util.isOnline(this.context)) {
566 Logger.e(TAG, "Ensure collector path is valid: %s", networkConnection.getUri());
567 }
568 Logger.e(TAG, "Emitter loop stopping: failures.");
569 isRunning.compareAndSet(true, false);
570 } else {
571 attemptEmit(getNetworkConnection()); // refresh network connection for next emit
572 }
573 }

586 @NonNull
587 protected List<Request> buildRequests(@NonNull List<EmitterEvent> events, HttpMethod httpMethod) {
588 List<Request> requests = new ArrayList<>();
589 String sendingTime = Util.getTimestamp();
590
591 if (httpMethod == GET) {
592 for (EmitterEvent event : events) {
593 Payload payload = event.payload;
594 addSendingTimeToPayload(payload, sendingTime);
595 boolean isOversize = isOversize(payload, httpMethod);
596 Request request = new Request(payload, event.eventId, isOversize);
597 requests.add(request);
598 }
599 } else {
600 for (int i = 0; i < events.size(); i += bufferOption.getCode()) {
601 List<Long> reqEventIds = new ArrayList<>();
602 List<Payload> postPayloadMaps = new ArrayList<>();
603
604 for (int j = i; j < (i + bufferOption.getCode()) && j < events.size(); j++) {
605 EmitterEvent event = events.get(j);
606 Payload payload = event.payload;
607 Long eventId = event.eventId;
608 addSendingTimeToPayload(payload, sendingTime);
609
610 if (isOversize(payload, httpMethod)) {
611 Request request = new Request(payload, eventId, true);
612 requests.add(request);
613
614 } else if (isOversize(payload, postPayloadMaps, httpMethod)) {
615 Request request = new Request(postPayloadMaps, reqEventIds);
616 requests.add(request);
617
618 // Clear collections and build a new POST
619 postPayloadMaps = new ArrayList<>();
620 reqEventIds = new ArrayList<>();
621
622 // Build and store the request
623 postPayloadMaps.add(payload);
624 reqEventIds.add(eventId);
625
626 } else {
627 postPayloadMaps.add(payload);
628 reqEventIds.add(eventId);
629 }
630 }
631
632 // Check if all payloads have been processed
633 if (!postPayloadMaps.isEmpty()) {
634 Request request = new Request(postPayloadMaps, reqEventIds);
635 requests.add(request);
636 }
637 }
638 }
639 return requests;
640 }

648 private boolean isOversize(@NonNull Payload payload, HttpMethod httpMethod) {
649 return isOversize(payload, new ArrayList<>(), httpMethod);

659 private boolean isOversize(@NonNull Payload payload, @NonNull List<Payload> previousPaylods, HttpMethod httpMethod) {
660 long byteLimit = httpMethod == GET ? byteLimitGet : byteLimitPost;
661 return isOversize(payload, byteLimit, previousPaylods);
662 }
663
671 private boolean isOversize(@NonNull Payload payload, long byteLimit, @NonNull List<Payload> previousPaylods) {
672 long totalByteSize = payload.getByteSize();
673 for (Payload previousPayload : previousPaylods) {
674 totalByteSize += previousPayload.getByteSize();
675 }
676 int wrapperBytes = previousPaylods.size() > 0 ? (previousPaylods.size() + POST_WRAPPER_BYTES) : 0;
677 return totalByteSize + wrapperBytes > byteLimit;
678 }
