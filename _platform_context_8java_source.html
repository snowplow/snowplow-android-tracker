68 @Nullable
69 public SelfDescribingJson getMobileContext(boolean userAnonymisation) {
70 updateEphemeralDictsIfNecessary();
71
72 // If does not contain the required properties, return null
73 if (!Util.mapHasKeys(pairs,
74 Parameters.OS_TYPE,
75 Parameters.OS_VERSION,
76 Parameters.DEVICE_MANUFACTURER,
77 Parameters.DEVICE_MODEL)) {
78 return null;
79 }
80
81 // If user anonymisation is on, remove the IDFA value
82 if (userAnonymisation && pairs.containsKey(Parameters.ANDROID_IDFA)) {
83 Map<String, Object> copy = new HashMap<>(pairs);
84 copy.remove(Parameters.ANDROID_IDFA);
85 return new SelfDescribingJson(TrackerConstants.MOBILE_SCHEMA, copy);
86 }
87
88 return new SelfDescribingJson(TrackerConstants.MOBILE_SCHEMA, pairs);
89 }

93 private synchronized void updateEphemeralDictsIfNecessary() {
94 long now = System.currentTimeMillis();
95 if (now - lastUpdatedEphemeralPlatformDict >= platformDictUpdateFrequency) {
96 setEphemeralPlatformDict();
97 }
98 if (now - lastUpdatedEphemeralNetworkDict >= networkDictUpdateFrequency) {
99 setEphemeralNetworkDict();
100 }
101 }

103 private void setPlatformDict() {
104 Util.addToMap(Parameters.OS_TYPE, deviceInfoMonitor.getOsType(), pairs);
105 Util.addToMap(Parameters.OS_VERSION, deviceInfoMonitor.getOsVersion(), pairs);
106 Util.addToMap(Parameters.DEVICE_MODEL, deviceInfoMonitor.getDeviceModel(), pairs);
107 Util.addToMap(Parameters.DEVICE_MANUFACTURER, deviceInfoMonitor.getDeviceVendor(), pairs);
108 Util.addToMap(Parameters.CARRIER, deviceInfoMonitor.getCarrier(context), pairs);
109 Util.addToMap(Parameters.PHYSICAL_MEMORY, deviceInfoMonitor.getPhysicalMemory(context), pairs);
110 Util.addToMap(Parameters.TOTAL_STORAGE, deviceInfoMonitor.getTotalStorage(), pairs);
111
112 setEphemeralPlatformDict();
113 setEphemeralNetworkDict();
114 }

116 private void setEphemeralPlatformDict() {
117 lastUpdatedEphemeralPlatformDict = System.currentTimeMillis();
118
119 Object currentIdfa = pairs.get(Parameters.ANDROID_IDFA);
120 if (currentIdfa == null || currentIdfa.toString().isEmpty()) {
121 Util.addToMap(Parameters.ANDROID_IDFA, deviceInfoMonitor.getAndroidIdfa(context), pairs);
122 }
123
124 Pair<String, Integer> batteryInfo = deviceInfoMonitor.getBatteryStateAndLevel(context);
125 if (batteryInfo != null) {
126 Util.addToMap(Parameters.BATTERY_STATE, batteryInfo.first, pairs);
127 Util.addToMap(Parameters.BATTERY_LEVEL, batteryInfo.second, pairs);
128 }
129 Util.addToMap(Parameters.SYSTEM_AVAILABLE_MEMORY, deviceInfoMonitor.getSystemAvailableMemory(context), pairs);
130 Util.addToMap(Parameters.AVAILABLE_STORAGE, deviceInfoMonitor.getAvailableStorage(), pairs);
131 }
132
133 private void setEphemeralNetworkDict() {
134 lastUpdatedEphemeralNetworkDict = System.currentTimeMillis();
135
136 NetworkInfo networkInfo = deviceInfoMonitor.getNetworkInfo(context);
137 Util.addToMap(Parameters.NETWORK_TECHNOLOGY, deviceInfoMonitor.getNetworkTechnology(networkInfo), pairs);
138 Util.addToMap(Parameters.NETWORK_TYPE, deviceInfoMonitor.getNetworkType(networkInfo), pairs);
139 }
