66 public State transition(@NonNull Event event, @Nullable State state) {
67 // - Init (DL) DeepLinkReceived
68 // - ReadyForOutput (DL) DeepLinkReceived
69 if (event instanceof DeepLinkReceived) {
70 DeepLinkReceived dlEvent = (DeepLinkReceived) event;
71 return new DeepLinkState(dlEvent.url, dlEvent.referrer);
72 } else {
73 // - Init (SV) Init
74 if (state == null) {
75 return null;
76 }
77 // - ReadyForOutput (SV) Init
78 DeepLinkState dlState = (DeepLinkState) state;
79 if (dlState.readyForOutput) {
80 return null;
81 }
82 // - DeepLinkReceived (SV) ReadyForOutput
83 DeepLinkState currentState = new DeepLinkState(dlState.url, dlState.referrer);
84 currentState.readyForOutput = true;
85 return currentState;
86 }
