408 private final NotificationCenter.FunctionalObserver receiveLifecycleNotification = new NotificationCenter.FunctionalObserver() {
409 @Override
410 public void apply(@NonNull Map<String, Object> data) {
411 Session session = getSession();
412 if (session == null || !lifecycleEvents) {
413 return;
414 }
415 Boolean isForeground = (Boolean) data.get("isForeground");
416 if (isForeground == null) {
417 return;
418 }
419 if (session.isBackground() == !isForeground) {
420 // if the new lifecycle state confirms the session state, there isn't any lifecycle transition
421 return;
422 }
423 if (isForeground) {
424 track(new Foreground().foregroundIndex(session.getForegroundIndex() + 1));
425 } else {
426 track(new Background().backgroundIndex(session.getBackgroundIndex() + 1));
427 }
428 session.setBackground(!isForeground);
429 }

617 public UUID track(final @NonNull Event event) {
618 if (!dataCollection.get()) {
619 return null;
620 }
621 event.beginProcessing(this);
622 TrackerStateSnapshot stateSnapshot;
623 TrackerEvent trackerEvent;
624 synchronized (this) {
625 stateSnapshot = stateManager.trackerStateForProcessedEvent(event);
626 trackerEvent = new TrackerEvent(event, stateSnapshot);
627 workaroundForIncoherentSessionContext(trackerEvent);
628 }
629 boolean reportsOnDiagnostic = !(event instanceof TrackerError);
630 Executor.execute(reportsOnDiagnostic, TAG, () -> {
631 transformEvent(trackerEvent);
632 Payload payload = payloadWithEvent(trackerEvent);
633 Logger.v(TAG, "Adding new payload to event storage: %s", payload);
634 this.emitter.add(payload);
635 event.endProcessing(this);
636 });
637 return trackerEvent.eventId;
638 }

640 private void transformEvent(@NonNull TrackerEvent event) {
641 // Application_install event needs the timestamp to the real installation event.
642 if (event.schema != null
643 && event.schema.equals(TrackerConstants.SCHEMA_APPLICATION_INSTALL)
644 && event.trueTimestamp != null)
645 {
646 event.timestamp = event.trueTimestamp;
647 event.trueTimestamp = null;
648 }
649 // Payload can be optionally updated with values based on internal state
650 stateManager.addPayloadValuesToEvent(event);
651 }
652
653 private @NonNull Payload payloadWithEvent(@NonNull TrackerEvent event) {
654 TrackerPayload payload = new TrackerPayload();
655 addBasicPropertiesToPayload(payload, event);
656 if (event.isPrimitive) {
657 addPrimitivePropertiesToPayload(payload, event);
658 } else {
659 addSelfDescribingPropertiesToPayload(payload, event);
660 }
661 List<SelfDescribingJson> contexts = event.contexts;
662 addBasicContextsToContexts(contexts, event);
663 addGlobalContextsToContexts(contexts, event);
664 addStateMachineEntitiesToContexts(contexts, event);
665 wrapContextsToPayload(payload, contexts);
666 return payload;
667 }

669 private void addBasicPropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
670 payload.add(Parameters.EID, event.eventId.toString());
671 payload.add(Parameters.DEVICE_TIMESTAMP, Long.toString(event.timestamp));
672 if (event.trueTimestamp != null) {
673 payload.add(Parameters.TRUE_TIMESTAMP, event.trueTimestamp.toString());
674 }
675 payload.add(Parameters.APPID, this.appId);
676 payload.add(Parameters.NAMESPACE, this.namespace);
677 payload.add(Parameters.TRACKER_VERSION, this.trackerVersion);
678 if (this.subject != null) {
679 payload.addMap(new HashMap<>(this.subject.getSubject()));
680 }
681 payload.add(Parameters.PLATFORM, this.devicePlatform.getValue());
682 }

684 private void addPrimitivePropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
685 payload.add(Parameters.EVENT, event.eventName);
686 payload.addMap(event.payload);
687 }
688
689 private void addSelfDescribingPropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
690 payload.add(Parameters.EVENT, TrackerConstants.EVENT_UNSTRUCTURED);
691
692 workaroundForCampaignAttributionEnrichment(payload, event); // TODO: To remove when Atomic table refactoring is finished
693
694 SelfDescribingJson data = new SelfDescribingJson(event.schema, event.payload);
695 HashMap<String, Object> unstructuredEventPayload = new HashMap<>();
696 unstructuredEventPayload.put(Parameters.SCHEMA, TrackerConstants.SCHEMA_UNSTRUCT_EVENT);
697 unstructuredEventPayload.put(Parameters.DATA, data.getMap());
698 payload.addMap(unstructuredEventPayload, base64Encoded, Parameters.UNSTRUCTURED_ENCODED, Parameters.UNSTRUCTURED);
699 }

701 /*
702 This is needed because the campaign-attribution-enrichment (in the pipeline) is able to parse
703 the `url` and `referrer` only if they are part of a PageView event.
704 The PageView event is an atomic event but the DeepLinkReceived is a SelfDescribing event.
705 For this reason we copy these two fields in the atomic fields in order to let the enrichment
706 to process correctly the fields even if the event is not a PageView and it's a SelfDescribing event.
707 This is a hack that should be removed once the atomic event table is dismissed and all the events
708 will be SelfDescribing.
709 */
710 private void workaroundForCampaignAttributionEnrichment(@NonNull Payload payload, @NonNull TrackerEvent event) {
711 if (event.schema.equals(DeepLinkReceived.SCHEMA) && event.payload != null) {
712 String url = (String)event.payload.get(DeepLinkReceived.PARAM_URL);
713 String referrer = (String)event.payload.get(DeepLinkReceived.PARAM_REFERRER);
714 payload.add(Parameters.PAGE_URL, url);
715 payload.add(Parameters.PAGE_REFR, referrer);
716 }

719 /*
720 The session context should be computed as part of `addBasicContextsToContexts` but that method
721 is executed in a separate thread. At the moment, the Session management is performed by the legacy
722 solution that doesn't make use of the tracker state snapshot (taken as soon as the event is tracked)
723 needed to keep a coherent state between events tracked in sequence but processed in parallel.
724 Due to this parallel processing the session context could be incoherent with the event sequence.
725 This limit is a serious problem when we process lifecycle events because the `application_background`
726 event should be checked with the session foreground timeout and the `application_foreground` event
727 should be checked with the session background timeout. Without a coherent session management it's
728 impossible to grant this behaviour causing serious issues on the calculation of the session expiring.
729 */
730 private void workaroundForIncoherentSessionContext(@NonNull TrackerEvent event) {
731 if (!event.isService && sessionContext) {
732 String eventId = event.eventId.toString();
733 long eventTimestamp = event.timestamp;
734 Session sessionManager = trackerSession;
735 if (sessionManager == null) {
736 Logger.track(TAG, "Session not ready or method getHasLoadedFromFile returned false with eventId: %s", eventId);
737 return;
738 }
739 SelfDescribingJson sessionContextJson = sessionManager.getSessionContext(eventId, eventTimestamp, userAnonymisation);
740 event.contexts.add(sessionContextJson);
741 }
742 }

744 private void addBasicContextsToContexts(@NonNull List<SelfDescribingJson> contexts, @NonNull TrackerEvent event) {
745 if (applicationContext) {

750 contexts.add(platformContext.getMobileContext(userAnonymisation));

754 return;

758 contexts.add(Util.getGeoLocationContext(this.context));
759 }

762 contexts.add(gdpr.getContext());

779 private void wrapContextsToPayload(@NonNull Payload payload, @NonNull List<SelfDescribingJson> contexts) {
780 if (contexts.isEmpty()) {
781 return;
782 }
783 List<Map<String, Object>> data = new LinkedList<>();
784 for (SelfDescribingJson context : contexts) {
785 if (context != null) {
786 data.add(context.getMap());
787 }
788 }
789 SelfDescribingJson finalContext = new SelfDescribingJson(TrackerConstants.SCHEMA_CONTEXTS, data);
790 payload.addMap(finalContext.getMap(), base64Encoded, Parameters.CONTEXT_ENCODED, Parameters.CONTEXT);
791 }

806 private void addServiceEventPayload(@NonNull Payload payload, @NonNull List<SelfDescribingJson> contexts) {
807 // Add default parameters to the payload
808 payload.add(Parameters.PLATFORM, this.devicePlatform.getValue());
809 payload.add(Parameters.APPID, this.appId);
810 payload.add(Parameters.NAMESPACE, this.namespace);
811 payload.add(Parameters.TRACKER_VERSION, this.trackerVersion);
812
813 // If there is a subject present for the Tracker add it
814 if (this.subject != null) {
815 payload.addMap(new HashMap<>(this.subject.getSubject()));
816 }
817
818 // Add Mobile Context
819 if (this.mobileContext) {
820 contexts.add(platformContext.getMobileContext(userAnonymisation));
821 }
822
823 // Add application context
824 if (this.applicationContext) {
825 contexts.add(Util.getApplicationContext(this.context));
826 }
827
828 // If there are contexts to nest
829 if (contexts.size() > 0) {
830 List<Map> contextMaps = new LinkedList<>();
831 for (SelfDescribingJson selfDescribingJson : contexts) {
832 if (selfDescribingJson != null) {
833 contextMaps.add(selfDescribingJson.getMap());
834 }
835 }
836 SelfDescribingJson envelope = new SelfDescribingJson(TrackerConstants.SCHEMA_CONTEXTS, contextMaps);
837 payload.addMap(envelope.getMap(), this.base64Encoded, Parameters.CONTEXT_ENCODED,
838 Parameters.CONTEXT);
839 }
840
841 // Add this payload to the emitter
842 this.emitter.add(payload);
843 }
