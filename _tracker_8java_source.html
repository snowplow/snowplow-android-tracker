395 private final NotificationCenter.FunctionalObserver receiveLifecycleNotification = new NotificationCenter.FunctionalObserver() {
396 @Override
397 public void apply(@NonNull Map<String, Object> data) {
398 Session session = getSession();
399 if (session == null || !lifecycleEvents) {
400 return;
401 }
402 Boolean isForeground = (Boolean) data.get("isForeground");
403 if (isForeground == null) {
404 return;
405 }
406 if (session.isBackground() == !isForeground) {
407 // if the new lifecycle state confirms the session state, there isn't any lifecycle transition
408 return;
409 }
410 if (isForeground) {
411 track(new Foreground().foregroundIndex(session.getForegroundIndex() + 1));
412 } else {
413 track(new Background().backgroundIndex(session.getBackgroundIndex() + 1));
414 }
415 session.setBackground(!isForeground);
416 }
417 };

507 if (trackerVersionSuffix != null) {
508 String suffix = trackerVersionSuffix.replaceAll("[^A-Za-z0-9.-]", "");

512 }
513
514 if (trackerDiagnostic && (level == LogLevel.OFF)) {
515 level = LogLevel.ERROR;
516 }
517
518 Logger.updateLogLevel(level);
519
520 // When session context is enabled
521 if (this.sessionContext) {
522 Runnable[] callbacks = {null, null, null, null};
523 if (sessionCallbacks.length == 4) {
524 callbacks = sessionCallbacks;
525 }
526 trackerSession = Session.getInstance(context, foregroundTimeout, backgroundTimeout, timeUnit, namespace, callbacks);

602 public void track(final @NonNull Event event) {
603 if (!dataCollection.get()) {
604 return;
605 }
606 event.beginProcessing(this);
607 TrackerStateSnapshot stateSnapshot;
608 TrackerEvent trackerEvent;
609 synchronized (this) {
610 stateSnapshot = stateManager.trackerStateForProcessedEvent(event);
611 trackerEvent = new TrackerEvent(event, stateSnapshot);
612 workaroundForIncoherentSessionContext(trackerEvent);
613 }
614 boolean reportsOnDiagnostic = !(event instanceof TrackerError);
615 Executor.execute(reportsOnDiagnostic, TAG, () -> {
616 transformEvent(trackerEvent);
617 Payload payload = payloadWithEvent(trackerEvent);
618 Logger.v(TAG, "Adding new payload to event storage: %s", payload);
619 this.emitter.add(payload);
620 event.endProcessing(this);
621 });
622 }

624 private void transformEvent(@NonNull TrackerEvent event) {
625 // Application_install event needs the timestamp to the real installation event.
626 if (event.schema != null
627 && event.schema.equals(TrackerConstants.SCHEMA_APPLICATION_INSTALL)
628 && event.trueTimestamp != null)
629 {
630 event.timestamp = event.trueTimestamp;
631 event.trueTimestamp = null;
632 }
633 // Payload can be optionally updated with values based on internal state
634 stateManager.addPayloadValuesToEvent(event);
635 }
636
637 private @NonNull Payload payloadWithEvent(@NonNull TrackerEvent event) {
638 TrackerPayload payload = new TrackerPayload();
639 addBasicPropertiesToPayload(payload, event);
640 if (event.isPrimitive) {
641 addPrimitivePropertiesToPayload(payload, event);
642 } else {
643 addSelfDescribingPropertiesToPayload(payload, event);
644 }
645 List<SelfDescribingJson> contexts = event.contexts;
646 addBasicContextsToContexts(contexts, event);
647 addGlobalContextsToContexts(contexts, event);
648 addStateMachineEntitiesToContexts(contexts, event);
649 wrapContextsToPayload(payload, contexts);
650 return payload;
651 }

653 private void addBasicPropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
654 payload.add(Parameters.EID, event.eventId.toString());
655 payload.add(Parameters.DEVICE_TIMESTAMP, Long.toString(event.timestamp));
656 if (event.trueTimestamp != null) {
657 payload.add(Parameters.TRUE_TIMESTAMP, event.trueTimestamp.toString());
658 }
659 payload.add(Parameters.APPID, this.appId);
660 payload.add(Parameters.NAMESPACE, this.namespace);
661 payload.add(Parameters.TRACKER_VERSION, this.trackerVersion);
662 if (this.subject != null) {
663 payload.addMap(new HashMap<>(this.subject.getSubject()));
664 }
665 payload.add(Parameters.PLATFORM, this.devicePlatform.getValue());
666 }

668 private void addPrimitivePropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
669 payload.add(Parameters.EVENT, event.eventName);
670 payload.addMap(event.payload);
671 }
672
673 private void addSelfDescribingPropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
674 payload.add(Parameters.EVENT, TrackerConstants.EVENT_UNSTRUCTURED);
675
676 workaroundForCampaignAttributionEnrichment(payload, event); // TODO: To remove when Atomic table refactoring is finished
677
678 SelfDescribingJson data = new SelfDescribingJson(event.schema, event.payload);
679 HashMap<String, Object> unstructuredEventPayload = new HashMap<>();
680 unstructuredEventPayload.put(Parameters.SCHEMA, TrackerConstants.SCHEMA_UNSTRUCT_EVENT);
681 unstructuredEventPayload.put(Parameters.DATA, data.getMap());
682 payload.addMap(unstructuredEventPayload, base64Encoded, Parameters.UNSTRUCTURED_ENCODED, Parameters.UNSTRUCTURED);
683 }

685 /*
686 This is needed because the campaign-attribution-enrichment (in the pipeline) is able to parse
687 the `url` and `referrer` only if they are part of a PageView event.
688 The PageView event is an atomic event but the DeepLinkReceived is a SelfDescribing event.
689 For this reason we copy these two fields in the atomic fields in order to let the enrichment
690 to process correctly the fields even if the event is not a PageView and it's a SelfDescribing event.
691 This is a hack that should be removed once the atomic event table is dismissed and all the events
692 will be SelfDescribing.
693 */
694 private void workaroundForCampaignAttributionEnrichment(@NonNull Payload payload, @NonNull TrackerEvent event) {
695 if (event.schema.equals(DeepLinkReceived.SCHEMA) && event.payload != null) {
696 String url = (String)event.payload.get(DeepLinkReceived.PARAM_URL);
697 String referrer = (String)event.payload.get(DeepLinkReceived.PARAM_REFERRER);
698 payload.add(Parameters.PAGE_URL, url);
699 payload.add(Parameters.PAGE_REFR, referrer);
700 }

703 /*
704 The session context should be computed as part of `addBasicContextsToContexts` but that method
705 is executed in a separate thread. At the moment, the Session management is performed by the legacy
706 solution that doesn't make use of the tracker state snapshot (taken as soon as the event is tracked)
707 needed to keep a coherent state between events tracked in sequence but processed in parallel.
708 Due to this parallel processing the session context could be incoherent with the event sequence.
709 This limit is a serious problem when we process lifecycle events because the `application_background`
710 event should be checked with the session foreground timeout and the `application_foreground` event
711 should be checked with the session background timeout. Without a coherent session management it's
712 impossible to grant this behaviour causing serious issues on the calculation of the session expiring.
713 */
714 private void workaroundForIncoherentSessionContext(@NonNull TrackerEvent event) {
715 if (!event.isService && sessionContext) {
716 String eventId = event.eventId.toString();
717 long eventTimestamp = event.timestamp;
718 Session sessionManager = trackerSession;
719 if (sessionManager == null) {
720 Logger.track(TAG, "Session not ready or method getHasLoadedFromFile returned false with eventId: %s", eventId);
721 return;
722 }
723 SelfDescribingJson sessionContextJson = sessionManager.getSessionContext(eventId, eventTimestamp);
724 event.contexts.add(sessionContextJson);
725 }
726 }

728 private void addBasicContextsToContexts(@NonNull List<SelfDescribingJson> contexts, @NonNull TrackerEvent event) {
729 if (applicationContext) {
730 contexts.add(Util.getApplicationContext(this.context));
731 }
732
733 if (mobileContext) {
734 contexts.add(platformContext.getMobileContext());
735 }
736
737 if (event.isService) {
738 return;
739 }
740
741 if (geoLocationContext) {
742 contexts.add(Util.getGeoLocationContext(this.context));
743 }
744
745 if (gdpr != null) {
746 contexts.add(gdpr.getContext());
747 }
748 }

751 synchronized (globalContextGenerators) {
752 for (GlobalContext generator : globalContextGenerators.values()) {
753 contexts.addAll(generator.generateContexts(event));

758 private void addStateMachineEntitiesToContexts(@NonNull List<SelfDescribingJson> contexts, @NonNull InspectableEvent event) {
759 List<SelfDescribingJson> stateManagerEntities = stateManager.entitiesForProcessedEvent(event);
760 contexts.addAll(stateManagerEntities);

763 private void wrapContextsToPayload(@NonNull Payload payload, @NonNull List<SelfDescribingJson> contexts) {
764 if (contexts.isEmpty()) {
765 return;
766 }
767 List<Map<String, Object>> data = new LinkedList<>();
768 for (SelfDescribingJson context : contexts) {
769 if (context != null) {
770 data.add(context.getMap());
771 }
772 }
773 SelfDescribingJson finalContext = new SelfDescribingJson(TrackerConstants.SCHEMA_CONTEXTS, data);
774 payload.addMap(finalContext.getMap(), base64Encoded, Parameters.CONTEXT_ENCODED, Parameters.CONTEXT);
775 }

790 private void addServiceEventPayload(@NonNull Payload payload, @NonNull List<SelfDescribingJson> contexts) {
791 // Add default parameters to the payload
792 payload.add(Parameters.PLATFORM, this.devicePlatform.getValue());
793 payload.add(Parameters.APPID, this.appId);
794 payload.add(Parameters.NAMESPACE, this.namespace);
795 payload.add(Parameters.TRACKER_VERSION, this.trackerVersion);
796
797 // If there is a subject present for the Tracker add it
798 if (this.subject != null) {
799 payload.addMap(new HashMap<>(this.subject.getSubject()));
800 }
801
802 // Add Mobile Context
803 if (this.mobileContext) {
804 contexts.add(platformContext.getMobileContext());
805 }
806
807 // Add application context
808 if (this.applicationContext) {
809 contexts.add(Util.getApplicationContext(this.context));
810 }
811
812 // If there are contexts to nest
813 if (contexts.size() > 0) {
814 List<Map> contextMaps = new LinkedList<>();
815 for (SelfDescribingJson selfDescribingJson : contexts) {
816 if (selfDescribingJson != null) {
817 contextMaps.add(selfDescribingJson.getMap());
818 }
819 }
820 SelfDescribingJson envelope = new SelfDescribingJson(TrackerConstants.SCHEMA_CONTEXTS, contextMaps);
821 payload.addMap(envelope.getMap(), this.base64Encoded, Parameters.CONTEXT_ENCODED,
822 Parameters.CONTEXT);
823 }
824
825 // Add this payload to the emitter
826 this.emitter.add(payload);
827 }
