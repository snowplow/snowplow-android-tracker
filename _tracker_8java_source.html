410 private final NotificationCenter.FunctionalObserver receiveLifecycleNotification = new NotificationCenter.FunctionalObserver() {
411 @Override
412 public void apply(@NonNull Map<String, Object> data) {
413 Session session = getSession();
414 if (session == null || !lifecycleEvents) {
415 return;
416 }
417 Boolean isForeground = (Boolean) data.get("isForeground");
418 if (isForeground == null) {
419 return;
420 }
421 if (session.isBackground() == !isForeground) {
422 // if the new lifecycle state confirms the session state, there isn't any lifecycle transition
423 return;
424 }
425 if (isForeground) {
426 track(new Foreground().foregroundIndex(session.getForegroundIndex() + 1));
427 } else {
428 track(new Background().backgroundIndex(session.getBackgroundIndex() + 1));
429 }
430 session.setBackground(!isForeground);
431 }

619 public UUID track(final @NonNull Event event) {
620 if (!dataCollection.get()) {
621 return null;
622 }
623 event.beginProcessing(this);
624 TrackerStateSnapshot stateSnapshot;
625 TrackerEvent trackerEvent;
626 synchronized (this) {
627 stateSnapshot = stateManager.trackerStateForProcessedEvent(event);
628 trackerEvent = new TrackerEvent(event, stateSnapshot);
629 workaroundForIncoherentSessionContext(trackerEvent);
630 }
631 boolean reportsOnDiagnostic = !(event instanceof TrackerError);
632 Executor.execute(reportsOnDiagnostic, TAG, () -> {
633 transformEvent(trackerEvent);
634 Payload payload = payloadWithEvent(trackerEvent);
635 Logger.v(TAG, "Adding new payload to event storage: %s", payload);
636 this.emitter.add(payload);
637 event.endProcessing(this);
638 });
639 return trackerEvent.eventId;
640 }

642 private void transformEvent(@NonNull TrackerEvent event) {
643 // Application_install event needs the timestamp to the real installation event.
644 if (event.schema != null
645 && event.schema.equals(TrackerConstants.SCHEMA_APPLICATION_INSTALL)
646 && event.trueTimestamp != null)
647 {
648 event.timestamp = event.trueTimestamp;
649 event.trueTimestamp = null;
650 }
651 // Payload can be optionally updated with values based on internal state
652 stateManager.addPayloadValuesToEvent(event);
653 }

655 private @NonNull Payload payloadWithEvent(@NonNull TrackerEvent event) {
656 TrackerPayload payload = new TrackerPayload();
657 addBasicPropertiesToPayload(payload, event);
658 if (event.isPrimitive) {
659 addPrimitivePropertiesToPayload(payload, event);
660 } else {
661 addSelfDescribingPropertiesToPayload(payload, event);
662 }
663 List<SelfDescribingJson> contexts = event.contexts;
664 addBasicContextsToContexts(contexts, event);
665 addGlobalContextsToContexts(contexts, event);
666 addStateMachineEntitiesToContexts(contexts, event);
667 wrapContextsToPayload(payload, contexts);
668 if (!event.isPrimitive) {
669 // TODO: To remove when Atomic table refactoring is finished
670 workaroundForCampaignAttributionEnrichment(payload, event, contexts);
671 }
672 return payload;
673 }

676 payload.add(Parameters.EID, event.eventId.toString());
677 payload.add(Parameters.DEVICE_TIMESTAMP, Long.toString(event.timestamp));

705 /*
706 This is needed because the campaign-attribution-enrichment (in the pipeline) is able to parse
707 the `url` and `referrer` only if they are part of a PageView event.
708 The PageView event is an atomic event but the DeepLinkReceived and ScreenView are SelfDescribing events.
709 For this reason we copy these two fields in the atomic fields in order to let the enrichment
710 to process correctly the fields even if the event is not a PageView and it's a SelfDescribing event.
711 This is a hack that should be removed once the atomic event table is dismissed and all the events
712 will be SelfDescribing.
713 */
714 private void workaroundForCampaignAttributionEnrichment(@NonNull Payload payload, @NonNull TrackerEvent event, List<SelfDescribingJson> contexts) {
715 String url = null;
716 String referrer = null;
717
718 if (event.schema.equals(DeepLinkReceived.SCHEMA) && event.payload != null) {
719 url = (String)event.payload.get(DeepLinkReceived.PARAM_URL);
720 referrer = (String)event.payload.get(DeepLinkReceived.PARAM_REFERRER);
721 }
722 else if (event.schema.equals(TrackerConstants.SCHEMA_SCREEN_VIEW) && contexts != null) {
723 for (SelfDescribingJson entity : contexts) {
724 if (entity instanceof DeepLink) {
725 DeepLink deepLink = (DeepLink) entity;
726 url = deepLink.getUrl();
727 referrer = deepLink.getReferrer();
728 break;
729 }
730 }
731 }
732
733 if (url != null) { payload.add(Parameters.PAGE_URL, url); }
734 if (referrer != null) { payload.add(Parameters.PAGE_REFR, referrer); }
735 }

737 /*
738 The session context should be computed as part of `addBasicContextsToContexts` but that method
739 is executed in a separate thread. At the moment, the Session management is performed by the legacy
740 solution that doesn't make use of the tracker state snapshot (taken as soon as the event is tracked)
741 needed to keep a coherent state between events tracked in sequence but processed in parallel.
742 Due to this parallel processing the session context could be incoherent with the event sequence.
743 This limit is a serious problem when we process lifecycle events because the `application_background`
744 event should be checked with the session foreground timeout and the `application_foreground` event
745 should be checked with the session background timeout. Without a coherent session management it's
746 impossible to grant this behaviour causing serious issues on the calculation of the session expiring.
747 */
748 private void workaroundForIncoherentSessionContext(@NonNull TrackerEvent event) {
749 if (!event.isService && sessionContext) {
750 String eventId = event.eventId.toString();
751 long eventTimestamp = event.timestamp;
752 Session sessionManager = trackerSession;
753 if (sessionManager == null) {
754 Logger.track(TAG, "Session not ready or method getHasLoadedFromFile returned false with eventId: %s", eventId);
755 return;
756 }
757 SelfDescribingJson sessionContextJson = sessionManager.getSessionContext(eventId, eventTimestamp, userAnonymisation);
758 event.contexts.add(sessionContextJson);
759 }
760 }
