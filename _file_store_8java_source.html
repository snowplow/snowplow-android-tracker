54 public static boolean saveMapToFile(@NonNull String filename, @NonNull Map objects, @NonNull Context context) {
55 FileOutputStream fos;
56 try {
57 Logger.d(TAG, "Attempting to save: %s", objects);
58 fos = context.openFileOutput(filename, Context.MODE_PRIVATE);
59 ObjectOutputStream oos = new ObjectOutputStream(fos);
60 oos.writeObject(objects);
61 oos.close();
62 Logger.d(TAG, " + Successfully saved KV Pairs to: %s", filename);
63 return true;
64 } catch(NullPointerException | IOException ioe){

78 @Nullable
79 public synchronized static Map<String, Object> getMapFromFile(@NonNull String filename, @NonNull Context context) {
80 try {
81 File file = context.getFileStreamPath(filename);
82 if (file == null || !file.exists()) {
83 return null;
84 }
85 Logger.d(TAG, "Attempting to retrieve map from: %s", filename);
86 FileInputStream fis = new FileInputStream(file);
87 ObjectInputStream ois = new ObjectInputStream(fis);
88 Map<String, Object> varsMap = (HashMap<String, Object>) ois.readObject();
89 ois.close();
90 Logger.d(TAG, " + Retrieved map from file: %s", varsMap);
91 return varsMap;
92 } catch (IOException | ClassNotFoundException ioe) {
93 Logger.e(TAG, " + Exception getting vars map: %s", ioe.getMessage());
94 }
95 return null;
