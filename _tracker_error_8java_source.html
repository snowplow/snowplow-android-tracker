56 HashMap<String, Object> payload = new HashMap<>();
57 payload.put(Parameters.DIAGNOSTIC_ERROR_CLASS_NAME, source);
58 payload.put(Parameters.DIAGNOSTIC_ERROR_MESSAGE, msg);

60 if (throwable != null) {
61 String stack = truncate(Util.stackTraceToString(throwable), MAX_STACK_LENGTH);
62 String throwableName = truncate(throwable.getClass().getName(), MAX_EXCEPTION_NAME_LENGTH);
63 payload.put(Parameters.DIAGNOSTIC_ERROR_STACK, stack);
64 payload.put(Parameters.DIAGNOSTIC_ERROR_EXCEPTION_NAME, throwableName);
65 }
