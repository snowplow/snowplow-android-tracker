71 @NonNull
72 public synchronized static List<String> removeUnsentEventsExceptForNamespaces(@NonNull Context context, @Nullable List<String> allowedNamespaces) {
73 if (allowedNamespaces == null) {
74 allowedNamespaces = new ArrayList<>();
75 }
76 String[] databaseList = context.databaseList();
77 if (databaseList == null) {
78 return new ArrayList<>();
79 }
80 List<String> allowedDbFiles = new ArrayList<>(allowedNamespaces.size());
81 for (String namespace : allowedNamespaces) {
82 String sqliteSuffix = namespace.replaceAll("[^a-zA-Z0-9_]+", "-");
83 String dbName = DATABASE_NAME + "-" + sqliteSuffix + ".sqlite";
84 allowedDbFiles.add(dbName);
85 allowedDbFiles.add(dbName + "-wal");
86 allowedDbFiles.add(dbName + "-shm");
87 }
88 List<String> removedDbFiles = new ArrayList<>();
89 for (String dbName : databaseList) {
90 if (!dbName.startsWith(DATABASE_NAME)) {
91 continue;
92 }
93 if (!allowedDbFiles.contains(dbName)) {
94 if (context.deleteDatabase(dbName)) {
95 removedDbFiles.add(dbName);
96 }
97 }
98 }
99 return removedDbFiles;
100 }

110 @NonNull

119 // Migrate old database if it exists
120 renameLegacyDatabase(context, dbName);

122 // Create database helper

124 instances.put(namespace, eventStoreHelper);

133 private static boolean renameLegacyDatabase(@NonNull Context context, String newDatabaseFilename) {
134 File database = context.getDatabasePath("snowplowEvents.sqlite");
135 File databaseWal = context.getDatabasePath("snowplowEvents.sqlite-wal");
136 File databaseShm = context.getDatabasePath("snowplowEvents.sqlite-shm");
137 File parentFile = database.getParentFile();
138 File newDatabase = new File(parentFile, newDatabaseFilename);
139 File newDatabaseWal = new File(parentFile, newDatabaseFilename + "-wal");
140 File newDatabaseShm = new File(parentFile, newDatabaseFilename + "-shm");
141 return database.renameTo(newDatabase)
142 && databaseWal.renameTo(newDatabaseWal)
143 && databaseShm.renameTo(newDatabaseShm);
144 }
