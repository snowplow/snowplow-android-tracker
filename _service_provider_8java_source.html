106 public ServiceProvider(@NonNull Context context, @NonNull String namespace, @NonNull NetworkConfiguration networkConfiguration, @NonNull List<Configuration> configurations) {
107 Objects.requireNonNull(context);
108 Objects.requireNonNull(networkConfiguration);
109 Objects.requireNonNull(configurations);
110 // Initialization
111 this.namespace = namespace;
112 this.context = context;
113 appId = context.getPackageName();
114 // Reset configurationUpdates
115 trackerConfigurationUpdate = new TrackerConfigurationUpdate(appId);
116 networkConfigurationUpdate = new NetworkConfigurationUpdate();

359 private Emitter makeEmitter() {

362 Emitter.EmitterBuilder builder = new Emitter.EmitterBuilder()

370 .byteLimitPost(emitterConfig.getByteLimitPost())
371 .byteLimitGet(emitterConfig.getByteLimitGet())
372 .threadPoolSize(emitterConfig.getThreadPoolSize())
373 .callback(emitterConfig.getRequestCallback())
374 .customRetryForStatusCodes(emitterConfig.getCustomRetryForStatusCodes());
375 HttpMethod method = networkConfig.getMethod();
376 if (method != null) {
377 builder.method(method);
378 }
379 Protocol protocol = networkConfig.getProtocol();
380 if (protocol != null) {
381 builder.security(protocol);
382 }
383 String endpoint = networkConfig.getEndpoint();
384 if (endpoint == null) {
385 endpoint = "";

388 if (emitterConfigurationUpdate.isPaused) {

394 @NonNull
395 private Tracker makeTracker() {
396 Emitter emitter = getEmitter();
397 Subject subject = getSubject();
398 TrackerConfigurationInterface trackerConfig = getTrackerConfigurationUpdate();
399 SessionConfigurationInterface sessionConfig = getSessionConfigurationUpdate();
400 Tracker.TrackerBuilder builder = new Tracker.TrackerBuilder(emitter, namespace, trackerConfig.getAppId(), context)
401 .subject(subject)
402 .trackerVersionSuffix(trackerConfig.getTrackerVersionSuffix())
403 .base64(trackerConfig.isBase64encoding())
404 .level(trackerConfig.getLogLevel())
405 .loggerDelegate(trackerConfig.getLoggerDelegate())
406 .platform(trackerConfig.getDevicePlatform())
407 .sessionContext(trackerConfig.isSessionContext())
408 .applicationContext(trackerConfig.isApplicationContext())
409 .mobileContext(trackerConfig.isPlatformContext())
410 .deepLinkContext(trackerConfig.isDeepLinkContext())
411 .screenContext(trackerConfig.isScreenContext())
412 .screenviewEvents(trackerConfig.isScreenViewAutotracking())
413 .lifecycleEvents(trackerConfig.isLifecycleAutotracking())
414 .installTracking(trackerConfig.isInstallAutotracking())
415 .applicationCrash(trackerConfig.isExceptionAutotracking())
416 .trackerDiagnostic(trackerConfig.isDiagnosticAutotracking())
417 .backgroundTimeout(sessionConfig.getBackgroundTimeout().convert(TimeUnit.SECONDS))
418 .foregroundTimeout(sessionConfig.getForegroundTimeout().convert(TimeUnit.SECONDS));
419 GdprConfigurationUpdate gdprConfig = getGdprConfigurationUpdate();
420 if (gdprConfig.sourceConfig != null) {
421 builder.gdprContext(
422 gdprConfig.getBasisForProcessing(),
423 gdprConfig.getDocumentId(),
424 gdprConfig.getDocumentVersion(),
425 gdprConfig.getDocumentDescription());
426 }
427 Tracker tracker = new Tracker(builder);
428 if (globalContextsConfiguration != null) {
429 tracker.setGlobalContextGenerators(globalContextsConfiguration.contextGenerators);
430 }
431 if (trackerConfigurationUpdate.isPaused) {
432 tracker.pauseEventTracking();
433 }
434 if (sessionConfigurationUpdate.isPaused) {
435 tracker.pauseSessionChecking();
436 }
437 Session session = tracker.getSession();
438 if (session != null) {
439 Consumer<SessionState> onSessionUpdate = sessionConfigurationUpdate.getOnSessionUpdate();
440 if (onSessionUpdate != null) {
441 session.onSessionUpdate = onSessionUpdate;
442 }
443 }
444 return tracker;
445 }
