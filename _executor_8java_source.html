70 public static void execute(boolean reportsOnDiagnostic, @Nullable String tag, @Nullable Runnable runnable) {
71 final String loggerTag;
72 if (tag == null) {
73 loggerTag = "Source not provided";
74 } else {
75 loggerTag = tag;
76 }
77 execute(runnable, t -> {
78 String message = t.getLocalizedMessage();
79 if (message == null) {
80 message = "No message provided.";
81 }
82 if (reportsOnDiagnostic) {
83 Logger.track(loggerTag, message, t);
84 } else {
85 Logger.e(loggerTag, message, t);

96 public static void execute(@Nullable Runnable runnable, @Nullable ExceptionHandler exceptionHandler) {
97 ExecutorService executor = getExecutor();
98 try {
99 executor.execute(() -> {
100 try {
101 if (runnable != null) {
102 runnable.run();
103 }
104 } catch (Throwable t) {
105 if (exceptionHandler != null) {
106 exceptionHandler.handle(t);
107 }
108 }
109 });
110 } catch (Exception e) {
111 if (exceptionHandler != null) {
112 exceptionHandler.handle(e);
113 }
114 }
