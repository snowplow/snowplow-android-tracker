215 Class<? extends Activity> activityClass = activity.getClass();
216 try {

218 Object reflectedValue = field.get(activity);

223 }
224 } catch (NoSuchFieldException e) {
225 Logger.d(TAG, String.format("Field `snowplowScreenId` not found on Activity `%s`.", activityClass.getSimpleName()), e);
226 } catch (Exception e) {
